// Descripion:
// Return true if the string in the first element of the array contains all of the letters of the string in the second element of the array.

// For example, ["hello", "Hello"], should return true because all of the letters in the second string are present in the first, ignoring case.

// The arguments ["hello", "hey"] should return false because the string hello does not contain a y.

// Lastly, ["Alien", "line"], should return true because all of the letters in line are present in Alien.

function mutation(arr) {
  let firstElement = arr[0].toLowerCase();
  let secondElement = arr[1].toLowerCase();

  for(let i=0;i<secondElement.length;i++){
    let letter = secondElement[i].toLowerCase();
    if(firstElement.indexOf(letter) === -1){
      return false;
    }
  }
  return true;
}

// output:

// Passed:4. mutation(["Mary", "Army"]) should return true.

// Passed:7. mutation(["floor", "for"]) should return true.

// Passed:10. mutation(["ate", "date"]) should return false.
// Passed:11. mutation(["Tiger", "Zebra"]) should return false.
// Passed:12. mutation(["Noel", "Ole"]) should ret

mutation(['hello', 'hey']);
