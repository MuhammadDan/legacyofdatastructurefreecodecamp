Problem Statement: Aapko ek function banana hai jo do cheezein accept kare:
- Ek array (e.g. [1, 2, 3, 4, 5])
- Ek number size (e.g. 2)
Aapka goal hai ke array ko size ke hisaab se chhoti arrays mein split karo.
Example:
chunkArray([1, 2, 3, 4, 5], 2);
// Output: [[1, 2], [3, 4], [5]]


** start of script.js **

// Question: Write a function that splits an array (first argument) into groups the length of size (second argument) and returns them as a two-dimensional array.

function chunkArrayInGroups(arr, size) {
  let result = [];
  
  for(var i=0;i<arr.length;i+=size){
    result.push(arr.slice(i,i+size))
  }
  return result;
}

console.log(chunkArrayInGroups(['a', 'b', 'c', 'd'], 2));
